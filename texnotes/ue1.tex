	v = \begin{pmatrix*}0\\2\\1\end{pmatrix*}\\
	v = v_{\bot} + v_{\parallel}\\\\
\end{split}\end{equation}
\begin{equation}\begin{split}
	<v; u>
	&= <v_{\bot} + v_{\parallel}; u>\\
	&= <v_{\bot}; u> + <v_{\parallel}; u>\\
	&= \cancelto{0}{<v_{\bot}; u>} \quad + <v_{\parallel}; u>\\
	&= <v_{\parallel}; u>\\
\end{split}\end{equation}
